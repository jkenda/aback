 3 (is the number a prime number?)
 4 macro is_prime?
 5     int -> bool
 6 is
 7 take candidate in
 8     2 true while <= * dup over candidate do
 9         take num prime in
10             + 1 num
11             if = 0 % candidate num then
12                 false
13             else prime end
14         end
15     end drop
16 end end
